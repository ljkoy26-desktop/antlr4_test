# 목표
  std::string 형태의 문장이 들어오면 
  해당문장이 db 상관없이 update, insert,delete, merge, select 인지 구분해주는 로직 만들어줘

# 신규 함수명
    bool isUpdateStmt(std::string s);
    bool isInsertStmt(std::string s);
    bool isDeleteStmt(std::string s);
    bool isMergeStmt(std::string s);
    bool isSelectStmt(std::string s);
